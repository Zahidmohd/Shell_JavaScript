# Test commands for brace expansion functionality
# Copy and paste these into the shell REPL

# Test 1: Basic comma-separated list
echo {a,b,c}

# Test 2: Brace expansion with prefix and suffix
echo file{1,2,3}.txt

# Test 3: Numeric sequence
echo {1..5}

# Test 4: Reverse numeric sequence
echo {5..1}

# Test 5: Character sequence
echo {a..e}

# Test 6: Reverse character sequence
echo {z..u}

# Test 7: Uppercase character sequence
echo {A..F}

# Test 8: Leading zeros in numeric sequence
echo {01..10}

# Test 9: Multiple brace expansions
echo {a,b}{1,2}

# Test 10: Brace expansion in commands
mkdir -p test_{dir1,dir2,dir3}

# Test 11: Complex brace expansion
echo prefix_{start,middle,end}_suffix

# Test 12: Nested prefix/suffix
echo test{x,y,z}file{1,2}.txt

# Test 13: No expansion (invalid pattern)
echo {nocomma}

# Test 14: No expansion (in single quotes)
echo '{a,b,c}'

# Test 15: Mixed with variables
echo $HOME/docs/{2024,2025}/{jan,feb}

# Test 16: With wildcards (note: * will be expanded by shell)
echo file{1..3}.*

# Test 17: Empty expansion element
echo {a,,c}

# Test 18: Longer sequences
echo {1..20}

# Test 19: Combined with other features
echo "Hello" {World,Universe}

# Test 20: Real-world example - backup files
echo backup_{2024..2026}_{01..12}.tar.gz
